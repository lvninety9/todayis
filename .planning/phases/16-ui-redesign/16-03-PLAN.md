```tsx
import { DndContext, closestCenter } from '@dnd-kit/core';
import { SortableContext, verticalListSortingStrategy } from '@dnd-kit/sortable';

<DndContext collisionDetection={closestCenter} onDragEnd={handleDragEnd}>

    {elements.map(el => (
      <SortableElement key={el.id} id={el.id}>
        <ElementCard element={el} />
      </SortableElement>
    ))}
  </SortableContext>
</DndContext>
```

```tsx
<div className="fixed bottom-4 left-1/2 -translate-x-1/2 bg-white dark:bg-gray-900 rounded-full shadow-lg px-4 py-2">

</div>
```

```tsx
<div className="flex flex-col lg:flex-row">
  <div className="w-full lg:w-1/3"> {/* Editor panel */}</div>

</div>
```

edit/page.tsx 툴바를 다음과 같이 재설계:

```tsx
const tools = [
  { icon: TextIcon, label: '텍스트', category: 'content' },
  { icon: ImageIcon, label: '이미지', category: 'media' },
  { icon: MoveIcon, label: '이동', category: 'layout' },
  { icon: UndoIcon, label: '실행취소', action: undo },
  { icon: RedoIcon, label: '다시실행', action: redo },
];

<FloatToolbar>
  {tools.map(tool => (
    <button onClick={tool.action} aria-label={tool.label}>
      <tool.icon />
      <span className="text-xs">{tool.label}</span>
    </button>
  ))}
</FloatToolbar>
```
